    Simple adding

    This function adds up all the numbers from 1 to user inputed number.
    For example: if the input is 4 then the program will return 10 because 1 + 2 + 3 + 4 = 10.

                            
                            function calculation(form) {
                                var inputNumber = form.inputBox.value;
                                var calculatedNumber = Number(inputNumber) * (Number(inputNumber) + 1) / 2
                                document.getElementById('result1').innerHTML = calculatedNumber;
                            };

    First factorial

    This function takes the user inputed number being passed and return the factorial of it.
    For example: if the input is 4 then the program will return 24, because 4 * 3 * 2 * 1 = 24.

                            
                            function firstFactorial(form) {
                                var num = form.inputBox.value;
                                var num = Number(num);
                                var result = 1;
                                for (var i = 2; i <= num; i++) {
                                    result *= i;
                                }
                                document.getElementById('result2').innerHTML = result;
                            };

    Longest word

    This function takes the inputed sentence being passed and returns the largest word in the string. If there are two or more words that are the same length, program returns the first word from the string with that length.
    For example: if the input is "I love studying at Fontys" then the program will return word "studying", because this word is largest in the sentence.

                            
                            function longestWord(form) {
                                var sentence = form.inputBox.value;
                                var str = sentence.split(" ");
                                var longest = 0;
                                var word = null;
                                str.forEach(function (str) {
                                    if (longest < str.length) {
                                        longest = str.length;
                                        word = str;
                                    }
                                });
                                var characters = word.length;
                                document.getElementById('result3').innerHTML = "Word " + word + " is " + characters + " characters long.";
                            };

    Word capitalization

    This function takes the inputed sentence being passed and capitalizes the first letter of each word. Words will be separated by only one space.
    For example: if the input is "I love studying at Fontys" then the program will return a sentence "I Love Studying At Fontys".

                            
                            function capitalize() {
                                var mySentence = document.getElementById('input5').value;
                                const words = mySentence.split(" ");
                
                                for (let i = 0; i < words.length; i++) {
                                    words[i] = words[i][0].toUpperCase() + words[i].substr(1);
                                }
                
                                finalWords = words.join(" ");
                
                                document.getElementById('capitalizeResult').innerHTML = finalWords;
                                console.log(finalWords)
                            };

    Rock Paper Scissors game


                            

                            <h1 id="info">Player 1 choose</h1>
                            <h2 id="answer"></h2>
                            <button id="rock">Rock</button>
                            <button id="paper">Paper</button>
                            <button id="scissors">Scissors</button>

    Explanation of HTML

    The id="info" and id="answer" are created to display which player turn it is and the winner of the game.
    Then buttons for the rock, paper and scissors are created with accompanying id's. 

                            
                                const optionR = document.getElementById("rock");
                                const optionP = document.getElementById("paper");
                                const optionS = document.getElementById("scissors");
                                const answer = document.getElementById('answer');
                                const info = document.getElementById('info');
                                var gamestate = 1;
                                var p1 = "";
                                var p2 = "";

    Explanation of JavaScript
    Variables for game state, player 1 and player 2 are created.

                            
                            optionR.addEventListener("click", function () {
                                gamestate == 1 ? (p1 = "Rock", gamestate++, info.innerHTML = "Player 2 choose", answer.innerHTML = "")
                                    : (p2 = "Rock", compareInputs(p1, p2), gamestate = 1, info.innerHTML = "Player 1 choose")
                            });

                            optionP.addEventListener("click", function () {
                                gamestate == 1 ? (p1 = "Paper", gamestate++, info.innerHTML = "Player 2 choose", answer.innerHTML = "")
                                    : (p2 = "Paper", compareInputs(p1, p2), gamestate = 1, info.innerHTML = "Player 1 choose")
                            });

                            optionS.addEventListener("click", function () {
                                gamestate == 1 ? (p1 = "Scissors", gamestate++, info.innerHTML = "Player 2 choose", answer.innerHTML = "")
                                    : (p2 = "Scissors", compareInputs(p1, p2), gamestate = 1, info.innerHTML = "Player 1 choose")
                            });

    A function for every button is created.
    When one of the buttons is clicked by player 1 the function sets the variable of p1 to the clicked button value, adds 1 to the game state and sets the info variable to say "Player 2 choose".
    When one of the buttons is clicked again the function sets the variable of p2 to the clicked button value. The compareInputs function is initiated with the updated p1 and p2 variables. Game state is set back to 1 and the info variable is set to say "Player 1 choose". 

                            
                            function compareInputs(p1, p2) {

                                (p1 === p2) ? answer.innerHTML = "It's a tie"
                                    : p1 === "Rock" && p2 === "Scissors" ? answer.innerHTML = "Player 1 wins"
                                        : p1 === "Paper" && p2 === "Rock" ? answer.innerHTML = "Player 1 wins"
                                            : p1 === "Scissors" && p2 === "Paper" ? answer.innerHTML = "Player 1 wins"
                                                : answer.innerHTML = "Player 2 wins";
                            };

    A function compareInputs is created which compares the input of player 1 and player 2.
    If both inputs are the same the answer is changed to "It's a tie".
    If first input is "Rock" and the second is "Scissors" then asnwer is changed to "Player 1 wins".
    Next two conditions are analogically simillar.
    Final condition is if none of the above conditions are true the answer is changed to "Player 2 wins".

    Tic Tac toe game


                            
                            <h1 class="game--title">Tic Tac Toe</h1>
                            <div class="game--container">
                            <div data-cell-index="0" class="cell"></div>
                            <div data-cell-index="1" class="cell"></div>
                            <div data-cell-index="2" class="cell"></div>
                            <div data-cell-index="3" class="cell"></div>
                            <div data-cell-index="4" class="cell"></div>
                            <div data-cell-index="5" class="cell"></div>
                            <div data-cell-index="6" class="cell"></div>
                            <div data-cell-index="7" class="cell"></div>
                            <div data-cell-index="8" class="cell"></div>
                            </div>
                            <h2 class="game--status"> </h2>
                            <button class="game--restart">Restart Game </button>

                        
                    
                            
                            const statusDisplay = document.querySelector('.game--status');

                            let gameActive = true;
                            let currentPlayer = "X";
                            let gameState = ["", "", "", "", "", "", "", "", ""];
                        
                            const winningMessage = () => `Player ${currentPlayer} has won!`;
                            const drawMessage = () => `Game ended in a draw!`;
                            const currentPlayerTurn = () => `It's ${currentPlayer}'s turn`;
                        
                            statusDisplay.innerHTML = currentPlayerTurn();
                        
                            const winningConditions = [
                                [0, 1, 2],
                                [3, 4, 5],
                                [6, 7, 8],
                                [0, 3, 6],
                                [1, 4, 7],
                                [2, 5, 8],
                                [0, 4, 8],
                                [2, 4, 6]
                            ];
                        
                    
                            
                            function handleCellPlayed(clickedCell, clickedCellIndex) {
                                gameState[clickedCellIndex] = currentPlayer;
                                clickedCell.innerHTML = currentPlayer;
                            }
                        
                    
                            
                            function handlePlayerChange() {
                                currentPlayer = currentPlayer === "X" ? "O" : "X";
                                statusDisplay.innerHTML = currentPlayerTurn();
                            }
                        
                    
                        
                        function handleResultValidation() {
                            let roundWon = false;
                            for (let i = 0; i <= 7; i++) {
                                const winCondition = winningConditions[i];
                                let a = gameState[winCondition[0]];
                                let b = gameState[winCondition[1]];
                                let c = gameState[winCondition[2]];
                                if (a === '' || b === '' || c === '') {
                                    continue;
                                }
                                if (a === b && b === c) {
                                    roundWon = true;
                                    break
                                }
                            }
                    
                            if (roundWon) {
                                statusDisplay.innerHTML = winningMessage();
                                gameActive = false;
                                return;
                            }
                    
                            let roundDraw = !gameState.includes("");
                            if (roundDraw) {
                                statusDisplay.innerHTML = drawMessage();
                                gameActive = false;
                                return;
                            }
                    
                            handlePlayerChange();
                        }
                    
                
                        
                        function handleCellClick(clickedCellEvent) {
                            const clickedCell = clickedCellEvent.target;
                            const clickedCellIndex = parseInt(clickedCell.getAttribute('data-cell-index'));
                    
                            if (gameState[clickedCellIndex] !== "" || !gameActive) {
                                return;
                            }
                    
                            handleCellPlayed(clickedCell, clickedCellIndex);
                            handleResultValidation();
                        }
                    
                
                        
                        function handleRestartGame() {
                            gameActive = true;
                            currentPlayer = "X";
                            gameState = ["", "", "", "", "", "", "", "", ""];
                            statusDisplay.innerHTML = currentPlayerTurn();
                            document.querySelectorAll('.cell').forEach(cell => cell.innerHTML = "");
                        }
                    
                        document.querySelectorAll('.cell').forEach(cell => cell.addEventListener('click', handleCellClick));
                        document.querySelector('.game--restart').addEventListener('click', handleRestartGame);
